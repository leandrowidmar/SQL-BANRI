
CREATE TABLE EXTRATO (
    ID_EXTRATO INTEGER PRIMARY KEY,
    NUMERO_CONTA INTEGER,
    NOME_TITULAR TEXT
);

CREATE TABLE TIPO_REGISTRO(
	ID INTEGER PRIMARY KEY,
  	TIPO INTEGER
);

CREATE TABLE REGISTRO(
	ID_REGISTRO INTEGER PRIMARY KEY,
    VALOR DECIMAL,
    TIPO_ID INTEGER,
  	ID_EXTRATO INTEGER,
  	FOREIGN KEY (ID_EXTRATO) REFERENCES EXTRATO(ID_EXTRATO),
    FOREIGN KEY (TIPO_ID) REFERENCES TIPO_REGISTRO(ID)
);

DROP TABLE REGISTRO;
DROP TABLE TIPO_REGISTRO;
DROP TABLE EXTRATO;
